        modulesExclude: ['ya-left-nav-', 'ya-question-breadcrumb-', 'ya-question-detail-', 'ya-q-774-textads-', 'ya-best-answer-', 'ya-qn-answers-', 'ya-qn-pagination-', 'ya-deleted-questions-', 'ya-useful-links-', 'ya-askbar-holder-', 'ya-darla-LREC-', 'ya-related-questions-', 'yai-q-answer-', 'ya-trending-', 'ya-darla-LREC2-', 'ya-darla-FSRVY-', 'ya-darla-FOOT-', 'ya-darla-MAST-', 'ya-darla-LDRB-', 'yucsHead-', 'UH-'],
        maxWaitTime: 1000,
        generateHAR: true,
        canShowVisualReport: true,
        useNormalizeCoverage: true
      };
      YAFT.init(yaftConfig, function(data, error) {
        if (!error) {
          Y.log(data, 'info');
          var lrecLoadTime = undefined, ldrbLoadTime = undefined;
          var aftTime = Math.round(data.aft);
          var visuallyCompleteTime = Math.round(data.visuallyComplete);
          var bucketId = 'NOT_IN_BUCKET';
          Y.log('aftTime: ' + aftTime + '; visuallyComplete: ' + visuallyCompleteTime +
            '; bucketId: ' + bucketId, 'debug');
          if (data.modulesReport['ya-darla-LREC']) {
            lrecLoadTime = Math.round(data.modulesReport['ya-darla-LREC'].loadTime);
            Y.log('LREC: ' + lrecLoadTime, 'debug');
          }
          if (data.modulesReport['ya-darla-LDRB']) {
            ldrbLoadTime = Math.round(data.modulesReport['ya-darla-LDRB'].loadTime);
            Y.log('LDRB: ' + ldrbLoadTime, 'debug');
          }
          if (window.LH) {
            LH.record('AFT', { name: 'AFT', type: 'mark', startTime: aftTime, duration: 0});
            LH.record('AFT_question', { name: 'AFT_question', type: 'mark', startTime: aftTime, duration: 0});
            LH.record('VIC', {name: 'VIC', type: 'mark', startTime: visuallyCompleteTime, duration: 0});
            if (lrecLoadTime) {
              LH.record('AD_LREC', {name: 'AD_LREC', type: 'mark', startTime: lrecLoadTime, duration: 0});
            }
            if (ldrbLoadTime) {
              LH.record('AD_LDRB', {name: 'AD_LDRB', type: 'mark', startTime: ldrbLoadTime, duration: 0});
            }
            LH.tag('b', {val: bucketId});
            LH.beacon({clearMarks:false, clearMeasures: false, clearCustomEntries: true, clearTags: false});
            Y.log('LH beacon sent', 'debug');
          } else {
            Y.log('LH beaconing not supported', 'debug');
          }
        } else {
          Y.log('Error loading AFT: ' + error, 'error');
        }
      });

      var aDadNode = Y.one("#ya-darla-LREC");
      var aTadNode = Y.one(".TA");

      if ((aDadNode || aTadNode) && ANSWERS.locale === "en_US") {
        if ((aDadNode !== null && aDadNode.get('clientHeight') === 0) || (aTadNode !== null && aTadNode.get('clientHeight') === 0)) {
          var aTnNode = Y.one("#ya-trending-now");

          if (aTnNode) {
            aTnNode.removeClass('Dn');
            Y.fire(Y.YA.Rapid.Track.NOTIFY.CLICK, {sec:'trndnow', slk:'abpview', pos:1});
          }
        }
      }
      Y.all('.ya-localtime').each(function(node){
        var ts = parseInt(node.getAttribute('data-ts')),
            d, tzOffset;
        if(!isNaN(ts)){
          d = new Date();
          d.setTime(ts * 1000);
          tzOffset = -(d.getTimezoneOffset()) / 60;
          tzOffset = tzOffset >= 0 ? '+' + tzOffset : tzOffset;
          if(!isNaN(d.getTime())){
            node.setAttribute('title', d.toLocaleDateString() + ' ' + d.toLocaleTimeString() + ' UTC' + tzOffset);
          }
        }
      });
    });
